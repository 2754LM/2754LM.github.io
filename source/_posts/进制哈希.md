# 例题

洛谷3370
https://www.luogu.com.cn/problem/P3370
```c++
#include<bits/stdc++.h>
typedef unsigned long long ull;
using namespace std;

ull v[10010];
int n;
int P = 131; //进制

int fun(string s){
	ull H = 0; //开ull溢出无所谓
	for(int i = 0; i < s.size(); i++){
		H = H*P + (s[i] - 'a')*P;
	}
	return H;
}

int main(){
	ios::sync_with_stdio(0);
	cin.tie(0);
	cout.tie(0);
	cin>>n;
	string tmp;
	for(int i = 0; i < n; i++){
		cin>>tmp;
		v[i] = fun(tmp);
	}
	sort(v, v + n);
	int ans = 0;
	for(int i = 1; i < n; i++){
		if(v[i] == v[i-1]) ans++;
	}
	cout<<n-ans;
	return 0;
}
```
